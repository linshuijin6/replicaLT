## 🎯 核心改进

### 问题分析

原始代码 (train.py) 中，三模态训练消耗大量显存的原因：

```python
# 原始代码 - 同时计算三个模态
x_t_fdg = t * seg_fdg + (1 - t) * images
v_fdg_prediction = model(x=x_t_fdg, timesteps=time_embedding, context=fdg_index)
v_fdg = seg_fdg - images
loss_fdg = F.mse_loss(v_fdg.float(), v_fdg_prediction.float())

x_t_av45 = t * seg_av45 + (1 - t) * images
v_av45_prediction = model(x=x_t_av45, timesteps=time_embedding, context=av45_index)
v_av45 = seg_av45 - images
loss_av45 = F.mse_loss(v_av45.float(), v_av45_prediction.float())

x_t_tau = t * seg_tau + (1 - t) * images
v_tau_prediction = model(x=x_t_tau, timesteps=time_embedding, context=tau_index)
v_tau = seg_tau - images
loss_tau = F.mse_loss(v_tau.float(), v_tau_prediction.float())

loss = alpha * loss_fdg + beta * loss_av45 + gamma * loss_tau
loss.backward()  # ❌ 同时保存三个模态的梯度
```

**内存占用**: 模型需要同时保存3个前向传播的中间激活值，用于反向传播。

### 优化方案

```python
# 优化代码 - 顺序计算，立即释放
# === FDG ===
x_t_fdg = t * seg_fdg + (1 - t) * images
v_fdg_prediction = model(x=x_t_fdg, timesteps=time_embedding, context=fdg_index)
loss_fdg = F.mse_loss(v_fdg_prediction, seg_fdg - images)
loss_fdg.backward()  # ✅ 立即反向传播
del x_t_fdg, v_fdg_prediction  # ✅ 释放显存
torch.cuda.empty_cache()

# === AV45 ===
x_t_av45 = t * seg_av45 + (1 - t) * images
v_av45_prediction = model(x=x_t_av45, timesteps=time_embedding, context=av45_index)
loss_av45 = F.mse_loss(v_av45_prediction, seg_av45 - images)
loss_av45.backward()
del x_t_av45, v_av45_prediction
torch.cuda.empty_cache()

# === TAU ===
x_t_tau = t * seg_tau + (1 - t) * images
v_tau_prediction = model(x=x_t_tau, timesteps=time_embedding, context=tau_index)
loss_tau = F.mse_loss(v_tau_prediction, seg_tau - images)
loss_tau.backward()
del x_t_tau, v_tau_prediction
torch.cuda.empty_cache()

# 参数更新（梯度已累积）
optimizer.step()
```

**内存节省**: 每次只保存1个模态的中间值，内存降低约 **60-70%**

## ⚙️ 配置参数详解

### GPU配置

```python
# 单GPU（推荐用于24GB显卡）
config.device_ids = [0]

# 双GPU并行（推荐，最佳性价比）
config.device_ids = [0, 1]

# 三GPU并行（跳过GPU 1，使用0,2,3）
config.device_ids = [0, 2, 3]

# 四GPU并行
config.device_ids = [0, 1, 2, 3]
```

### 梯度累积

用于模拟更大的batch size，同时降低单步显存：

```python
# 显存充足，不需要累积
config.accumulation_steps = 1  # 相当于 bs=1

# 显存紧张，累积2步（推荐）
config.accumulation_steps = 2  # 相当于 bs=2，显存减半

# 显存非常紧张，累积4步
config.accumulation_steps = 4  # 相当于 bs=4，显存降至1/4
```

### 模态权重

根据临床重要性调整：

```python
# 默认：三个模态同等重要
config.alpha = 1.0   # FDG
config.beta = 1.0    # AV45
config.gamma = 1.0   # TAU

# 示例：更重视淀粉样蛋白（AV45）
config.alpha = 1.0
config.beta = 2.0    # ← 加倍
config.gamma = 1.0

# 示例：暂时不训练TAU
config.gamma = 0.0
```

## 🔧 常见问题解决

### Q1: 仍然显存溢出 (OOM)

**解决方案**（按优先级）：

1. **增加梯度累积**
   ```python
   config.accumulation_steps = 4  # 或更大
   ```

2. **启用混合精度**（如果还没启用）
   ```python
   config.use_amp = True
   ```

3. **减小输入尺寸**
   ```python
   # 在数据预处理中
   mt.ResizeWithPadOrCropd(keys=pet_keys, spatial_size=[128, 128, 128])
   # 原来是 [160, 192, 160]
   ```

4. **使用梯度检查点**（需修改模型定义）
   ```python
   from torch.utils.checkpoint import checkpoint
   
   # 在模型中
   def forward(self, x):
       x = checkpoint(self.block1, x)  # 用时间换空间
       x = checkpoint(self.block2, x)
       return x
   ```

### Q2: 多GPU训练速度反而变慢

**可能原因**：

1. **数据加载瓶颈**
   ```python
   # 增加数据加载进程
   train_loader = DataLoader(
       train_ds, 
       batch_size=1, 
       num_workers=4,  # ← 增加这个值
       pin_memory=True
   )
   ```

2. **通信开销大于收益**
   - 发生在：模型很小但GPU很多
   - 解决：减少GPU数量，或增大模型/batch size

3. **GPU负载不均**
   ```python
   # 检查各GPU使用率
   nvidia-smi dmon -s u
   ```

### Q3: 损失不收敛或NaN

**可能原因**：

1. **混合精度导致数值不稳定**
   ```python
   # 临时关闭混合精度测试
   config.use_amp = False
   ```

2. **梯度累积比例错误**
   ```python
   # 检查损失缩放
   loss = loss / config.accumulation_steps  # 必须缩放
   ```

3. **学习率过大**
   ```python
   # 降低学习率
   config.learning_rate = 1e-5  # 原来 2.5e-5
   ```

### 2. 添加保存和恢复

```python
def save_checkpoint(trainer, epoch, checkpoint_dir='./checkpoints'):
    os.makedirs(checkpoint_dir, exist_ok=True)
    
    checkpoint = {
        'epoch': epoch,
        'model_state_dict': trainer.model.state_dict(),
        'optimizer_state_dict': trainer.optimizer.state_dict(),
        'scaler_state_dict': trainer.scaler.state_dict(),
    }
    
    path = os.path.join(checkpoint_dir, f'checkpoint_epoch_{epoch}.pth')
    torch.save(checkpoint, path)
    print(f"✓ 保存检查点: {path}")

def load_checkpoint(trainer, checkpoint_path):
    checkpoint = torch.load(checkpoint_path)
    
    trainer.model.load_state_dict(checkpoint['model_state_dict'])
    trainer.optimizer.load_state_dict(checkpoint['optimizer_state_dict'])
    trainer.scaler.load_state_dict(checkpoint['scaler_state_dict'])
    
    start_epoch = checkpoint['epoch'] + 1
    print(f"✓ 加载检查点: epoch {checkpoint['epoch']}")
    return start_epoch

# 使用
# 训练中保存
if (epoch + 1) % 10 == 0:
    save_checkpoint(trainer, epoch)

# 恢复训练
# start_epoch = load_checkpoint(trainer, 'checkpoints/checkpoint_epoch_50.pth')
```

### 3. 早停和学习率调度

```python
from torch.optim.lr_scheduler import ReduceLROnPlateau

# 创建调度器
scheduler = ReduceLROnPlateau(
    trainer.optimizer, 
    mode='min',
    factor=0.5,
    patience=5,
    verbose=True
)

# 早停
best_val_loss = float('inf')
patience_counter = 0
max_patience = 10

for epoch in range(n_epochs):
    # 训练
    train_losses = train_epoch(trainer, train_loader, epoch)
    
    # 验证
    if (epoch + 1) % config.val_interval == 0:
        val_losses = validate_epoch(trainer, val_loader)
        
        # 学习率调度
        scheduler.step(val_losses['total'])
        
        # 早停
        if val_losses['total'] < best_val_loss:
            best_val_loss = val_losses['total']
            patience_counter = 0
            save_checkpoint(trainer, epoch, is_best=True)
        else:
            patience_counter += 1
            
        if patience_counter >= max_patience:
            print(f"早停: {max_patience} 个epoch内验证损失未改善")
            break
```

## 📝 完整示例

将优化方案集成到原有train.py的完整示例：

```python
# train_with_optimization.py
import torch
from train_multi_gpu_optimized import MemoryEfficientTrainer, TrainingConfig
from monai.data import PersistentDataset, DataLoader
import monai.transforms as mt

# === 1. 配置 ===
config = TrainingConfig()
config.device_ids = [0, 1]  # 你的GPU
config.accumulation_steps = 2
config.use_amp = True

# === 2. 数据加载（使用你原有的代码）===
from train import (
    mat_load, get_subject_id, FillMissingPET,
    fdg_index_transform, av45_index_transform, tau_index_transform
)

# 你的原有数据处理代码
train_json_path = "..."
val_json_path = "..."
# ... 加载 train_data, val_data ...

pet_keys = ["mri", "av45", "fdg", "tau"]
train_transforms = mt.Compose([
    mt.Lambdad(keys=pet_keys, func=mat_load),
    FillMissingPET(keys=pet_keys, ref_key="mri"),
    # ... 你的其他transforms ...
])

train_ds = PersistentDataset(data=train_data, transform=train_transforms, ...)
train_loader = DataLoader(train_ds, batch_size=1, shuffle=True, num_workers=2)

val_ds = PersistentDataset(data=val_data, transform=val_transforms, ...)
val_loader = DataLoader(val_ds, batch_size=1, shuffle=False, num_workers=2)

# === 3. 模型（使用你原有的模型）===
from generative.networks.nets import DiffusionModelUNet

model = DiffusionModelUNet(
    spatial_dims=3,
    in_channels=1,
    out_channels=1,
    num_channels=(32, 64, 64, 128),
    attention_levels=(False, False, False, True),
    num_res_blocks=1,
    num_head_channels=(0, 0, 0, 128),
    with_conditioning=True,
    cross_attention_dim=512,
    use_flash_attention=True,
)

# === 4. 创建优化训练器 ===
trainer = MemoryEfficientTrainer(model, config)

# === 5. 训练循环 ===
from tqdm import tqdm

for epoch in range(config.n_epochs):
    # 训练
    trainer.model.train()
    epoch_losses = {'fdg': 0.0, 'av45': 0.0, 'tau': 0.0, 'total': 0.0}
    
    progress_bar = tqdm(train_loader, desc=f'Epoch {epoch}')
    for step, data in enumerate(progress_bar):
        losses = trainer.train_step_sequential(data)
        
        for key in epoch_losses:
            epoch_losses[key] += losses[key]
        
        progress_bar.set_postfix({
            'loss': f'{epoch_losses["total"]/(step+1):.4f}'
        })
    
    # 打印epoch总结
    print(f"\nEpoch {epoch} 训练损失:")
    for key, val in epoch_losses.items():
        print(f"  {key}: {val/len(train_loader):.6f}")
    
    # 验证
    if (epoch + 1) % config.val_interval == 0:
        trainer.model.eval()
        val_losses = {'fdg': 0.0, 'av45': 0.0, 'tau': 0.0, 'total': 0.0}
        
        with torch.no_grad():
            for data in tqdm(val_loader, desc='Validation'):
                losses = trainer.validation_step(data)
                for key in val_losses:
                    val_losses[key] += losses[key]
        
        print(f"\nEpoch {epoch} 验证损失:")
        for key, val in val_losses.items():
            print(f"  {key}: {val/len(val_loader):.6f}")
        
        # 保存检查点
        checkpoint = {
            'epoch': epoch,
            'model': trainer.model.state_dict(),
            'optimizer': trainer.optimizer.state_dict(),
        }
        torch.save(checkpoint, f'checkpoint/epoch_{epoch}.pth')
        print(f"✓ 保存检查点")

print("\n训练完成！")
```
